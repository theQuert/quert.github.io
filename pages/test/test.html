from openai import OpenAI

client = OpenAI(
    base_url="https://api.stima.tech/v1/",
    api_key=key
)

response = client.chat.completions.create(
    # model="gpt-4o",
    # model="claude-3-5-sonnet-20240620",
    # model="gpt-3.5-turbo",
    model="cow-3-5-20240620",

        {"role": "user", "content": "Hi, this is user speaking. Say hi to me."},]

    )
print(response)
